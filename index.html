Determinazione della velocità del suono

Esperimento di fisica realizzato con Arduino e sensore ultrasuoni.

Codice Arduino (.ino) 
// Determinazione della velocità del suono mediante Arduino
int trigPin = 2;                           // Definizione Pin di trigger
int echoPin = 4;                           // Definizione Pin di Eco
int i = 0;                                 // definizione variabile del ciclo
int N = 50;                                // Quanti cicli di misurazione vogliamo?

void setup ()                              // inizializzazione del sensore
{
  Serial.begin (9600);                     // imposta la velocità di trasmissione
  pinMode(trigPin , OUTPUT );              // imposta questo pin come output
  pinMode(echoPin , INPUT );               // imposta questo pin come input
  digitalWrite(trigPin , LOW);             // mandiamo un segnale basso
}

void loop()                                // inizio del programma
{
  digitalWrite(trigPin , LOW);
  for (i = 1; i <= N; i++)                 // ciclo di ripetizione delle misure
  {
    digitalWrite(trigPin , HIGH);          // mandiamo il segnale sonoro
    delayMicroseconds (10);                // aspetta 10 microsecondi
    digitalWrite(trigPin , LOW );          // abbassa il segnale
    long tUs = pulseIn(echoPin , HIGH);    // ascolta l'eco
    delayMicroseconds (10);
    Serial.print(tUs);
    Serial.print("\n");                    
    delay (10);                            // pausa tra misurazioni
  }
  Serial.print("x\n");
  delay (10);                              
}
